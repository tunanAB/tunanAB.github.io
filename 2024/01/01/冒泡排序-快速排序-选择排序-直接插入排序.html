#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>

int i, j;

void Bubble_sort(int a[], int n)//冒泡排序，注意其中两次排序的终止次数。
{
	int temp;
	for (i = 0; i < n - 1; i++)
	{
		for (j = 0; j < n - 1 - i; j++)
		{
			if (a[j] > a[j + 1])
			{
				temp = a[j];
				a[j] = a[j + 1];
				a[j + 1] = temp;
			}
		}
	}
}

void InsertSort(int a[], int l)//直接插入排序
{
	int temp;
	int j;
	for (int i = 1; i < l; i++)
	{
		if (a[i] < a[i - 1])
		{
			temp = a[i];
			for (j = i - 1; j >= 0 && temp < a[j]; j--)
			{
				a[j + 1] = a[j];
			}
			a[j + 1] = temp;

		}
	}
}

void Select_sort(int a[], int n)
{
	int min,mi;
	for (i = 0; i < n; i++)
	{
		min = a[i];
		mi = i;
		for (j = i + 1; j < n; j++)
		{
			if (min > a[j])
			{
				min = a[j];
				mi = j;
			}
		}
		if (mi != i)
		{
			a[mi] = a[i];
			a[i] = min;
		}
	}
}

void Quick_sort(int a[], int l, int r)//快速排序，从小到大。
{
	if (l < r)
	{
		int i = l, j = r, x = a[l];
		while (i < j)
		{
			while (i < j && a[j] >= x)
				j--;
			if (i < j)
				a[i++] = a[j];

			while (i < j && a[i] < x)
				i++;
			if (i < j)
				a[j--] = a[i];
		}
		a[i] = x;
		Quick_sort(a, l, i - 1);
		Quick_sort(a, i + 1, r);
	}
} 

int main()
{
	int a[10];
	for (i = 0; i < 10; i++)
	{
		scanf("%d", &a[i]);
	}
	Bubble_sort(a, 10);
	for (i = 0; i < 10; i++)
	{
		printf("%d ", a[i]);
	}
	printf("\n");
	Select_sort(a, 10);
	for (i = 0; i < 10; i++)
	{
		printf("%d ", a[i]);
	}
	printf("\n");
	Quick_sort(a, 0, 9);
	for (i = 0; i < 10; i++)
	{
		printf("%d ", a[i]);
	}
	return 0;
}
